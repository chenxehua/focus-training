    utterance.lang = config.language;
    utterance.rate = config.rate;
    utterance.pitch = config.pitch;

// 迷宫寻路算法 - A*算法
interface Point {
  x: number;
  y: number;
}

interface Node {
  position: Point;
  g: number;  // 从起点到当前点的成本
  h: number;  // 启发式估计到目标的成本
  f: number;  // 总成本 g + h
  parent: Node | null;
}

function aStarPathfinding(
  start: Point,
  end: Point,
  maze: number[][]
): Point[] {
  const openSet: Node[] = [];
  const closedSet = Set<Point>();
  const startNode: Node = {
    position: start,
    g: 0,
    h: heuristic(start, end),
    f: 0,
    parent: null,
  };
  startNode.f = startNode.g + startNode.h;
  
  openSet.push(startNode);
  
  while (openSet.length > 0) {
    // 找到f值最小的节点
    openSet.sort((a, b) => a.f - b.f);
    const current = openSet.shift()!;
    
    // 到达目标
    if (current.position.x === end.x && current.position.y === end.y) {
      return reconstructPath(current);
    }
    
    closedSet.add(current.position);
    
    // 检查相邻节点
    const neighbors = getNeighbors(current.position, maze);
    for (const neighborPos of neighbors) {
      if (closedSet.has(neighborPos)) continue;
      
      const g = current.g + 1;
      const existingNode = openSet.find(
        n => n.position.x === neighborPos.x && n.position.y === neighborPos.y
      );
      
      if (existingNode && g < existingNode.g) {
        existingNode.g = g;
        existingNode.f = existingNode.g + existingNode.h;
        existingNode.parent = current;
      } else if (!existingNode) {
        const h = heuristic(neighborPos, end);
        openSet.push({
          position: neighborPos,
          g,
          h,
          f: g + h,
          parent: current,
        });
      }
    }
  }
  
  return []; // 无路径
}

// 启发式函数（曼哈顿距离）
function heuristic(a: Point, b: Point): number {
  return Math.abs(a.x - b.x) + Math.abs(a.y - b.y);
}

// 重建路径
function reconstructPath(node: Node): Point[] {
  const path: Point[] = [];
  let current: Node | null = node;
  
  while (current) {
    path.unshift(current.position);
    current = current.parent;
  }
  
  return path;
}

class ItemGenerator {
  private categories: string[];
  private items: Map<string, string[]>;  // category -> items
  
  constructor(categories: string[]) {
    this.categories = categories;
    this.items = new Map();
    categories.forEach(cat => {
      this.items.set(cat, SORTING_RULES[cat]);
    });
  }
  
  // 生成新物品
  generate(): FallingItem {
    const category = this.categories[Math.floor(Math.random() * this.categories.length)];
    const categoryItems = this.items.get(category)!;
    const itemName = categoryItems[Math.floor(Math.random() * categoryItems.length)];
    
    return {
      id: `${itemName}_${Date.now()}`,
      name: itemName,
      category,
      x: Math.random() * (screenWidth - 80),
      y: -60,
      targetY: screenHeight + 60,
      speed: 100 + Math.random() * 50,  // 100-150 像素/秒
      rotation: Math.random() * 360,
      rotationSpeed: (Math.random() - 0.5) * 100,
    };
  }
}

// 物品下落动画循环
function updateFallingItems(items: FallingItem[], deltaTime: number) {
  items.forEach(item => {
    // 更新Y位置
    item.y += item.speed * (deltaTime / 1000);
    
    // 更新旋转
    item.rotation += item.rotationSpeed * (deltaTime / 1000);
    
    // 检测是否超出屏幕底部
    if (item.y > item.targetY) {
      // 标记为错过
      itemWasMissed(item);
    }
  });
  
  // 移除超出屏幕的物品
  return items.filter(item => item.y <= item.targetY);
}

// 碰撞检测
function checkCollision(
  item: FallingItem,
  basket: Basket
): boolean {
  const itemBox = {
    x: item.x,
    y: item.y,
    width: 64,
    height: 64,
  };
  
  const basketBox = {
    x: basket.x,
    y: basket.y,
    width: 100,
    height: 80,
  };
  
  return (
    itemBox.x < basketBox.x + basketBox.width &&
    itemBox.x + itemBox.width > basketBox.x &&
    itemBox.y < basketBox.y + basketBox.height &&
    itemBox.y + itemBox.height > basketBox.y
  );
}

```typescript
// 星星跟踪系统
class StarTrackingSystem {
  private stars: Star[];
  private targetStar: Star;
  private canvas: HTMLCanvasElement;
  private ctx: CanvasRenderingContext2D;
  
  constructor(canvas: HTMLCanvasElement, starCount: number) {
    this.canvas = canvas;
    this.ctx = canvas.getContext('2d')!;
    this.stars = this.generateStars(starCount);
    this.targetStar = this.selectTargetStar();
  }
  
  // 生成星星
  private generateStars(count: number): Star[] {
    const stars: Star[] = [];
    const colors = ['#FFD93D', '#C0C0C0', '#87CEEB', '#FFB5C5', '#9B59B6'];
    
    for (let i = 0; i < count; i++) {
      const star: Star = {
        id: i,
        x: Math.random() * (this.canvas.width - 96) + 48,
        y: Math.random() * (this.canvas.height - 96) + 48,
        color: colors[Math.floor(Math.random() * colors.length)],
        size: 48,
        opacity: 1,
        blinkPattern: 'none',
        blinkTimer: 0,
        path: [],
        currentPathIndex: 0,
      };
      stars.push(star);
    }
    
    return stars;
  }
  
  // 选择目标星星
  private selectTargetStar(): Star {
    const randomIndex = Math.floor(Math.random() * this.stars.length);
    this.stars[randomIndex].isTarget = true;
    this.stars[randomIndex].blinkPattern = 'blink';
    return this.stars[randomIndex];
  }
  
  // 更新星星位置
  update(deltaTime: number) {
    this.stars.forEach(star => {
      // 移动星星
      if (star.path.length > 0 && star.currentPathIndex < star.path.length) {
        const targetPoint = star.path[star.currentPathIndex];
        
        // 插值移动
        const dx = targetPoint.x - star.x;
        const dy = targetPoint.y - star.y;
        const distance = Math.sqrt(dx * dx + dy * dy);
        const speed = 100; // 像素/秒
        const moveAmount = speed * (deltaTime / 1000);
        
        if (distance <= moveAmount) {
          star.x = targetPoint.x;
          star.y = targetPoint.y;
          star.currentPathIndex++;
        } else {
          star.x += (dx / distance) * moveAmount;
          star.y += (dy / distance) * moveAmount;
        }
      }
      
      // 闪烁效果
      if (star.blinkPattern !== 'none') {
        star.blinkTimer += deltaTime;
        const blinkInterval = star.blinkPattern === 'blink' ? 300 : 150;
        
        if (star.blinkTimer >= blinkInterval) {
          star.opacity = star.opacity === 1 ? 0.3 : 1;
          star.blinkTimer = 0;
        }
      }
    });
  }
  
  // 渲染
  render() {
    this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
    
    this.stars.forEach(star => {
      // 绘制光晕
      if (star.isTarget) {
        const gradient = this.ctx.createRadialGradient(
          star.x, star.y, star.size * 0.5,
          star.x, star.y, star.size * 2
        );
        gradient.addColorStop(0, 'rgba(255, 217, 61, 0.4)');
        gradient.addColorStop(1, 'rgba(255, 217, 61, 0)');
        this.ctx.fillStyle = gradient;
        this.ctx.fillRect(
          star.x - star.size * 2,
          star.y - star.size * 2,
          star.size * 4,
          star.size * 4
        );
      }
      
      // 绘制星星
      this.ctx.globalAlpha = star.opacity;
      this.drawStar(star.x, star.y, star.size, star.color);
      this.ctx.globalAlpha = 1;
    });
  }
  
  // 绘制星星形状
  private drawStar(x: number, y: number, size: number, color: string) {
    this.ctx.fillStyle = color;
    this.ctx.beginPath();
    
    const spikes = 5;
    const outerRadius = size / 2;
    const innerRadius = outerRadius / 2;
    
    for (let i = 0; i < spikes * 2; i++) {
      const radius = i % 2 === 0 ? outerRadius : innerRadius;
      const angle = (Math.PI / spikes) * i - Math.PI / 2;
      const px = x + Math.cos(angle) * radius;
      const py = y + Math.sin(angle) * radius;
      
      if (i === 0) {
        this.ctx.moveTo(px, py);
      } else {
        this.ctx.lineTo(px, py);
      }
    }
    
    this.ctx.closePath();
    this.ctx.fill();
  }
}

function calculateUniversalScore(params: {
  correctRate: number;
  speedBonus: number;
  difficulty: number;
  completionBonus?: number;
}): number {
  const { correctRate, speedBonus, difficulty, completionBonus } = params;
  
  const baseScore = correctRate * 100;
  const difficultyFactor = difficulty * 0.15 + 0.5;
  const total = (baseScore + speedBonus + (completionBonus || 0)) * difficultyFactor;
  
  return Math.round(Math.min(total, 999));
}

function calculateUniversalStars(
  score: number,
  difficulty: number
): number {
  const thresholds = {
    1: [30, 60, 90],
    2: [40, 70, 100],
    3: [50, 80, 110],
    4: [60, 90, 120],
    5: [70, 100, 130],
  };
  
  const levelThresholds = thresholds[difficulty as keyof typeof thresholds];
  
  if (score >= levelThresholds[2]) return 3;
  if (score >= levelThresholds[1]) return 2;
  if (score >= levelThresholds[0]) return 1;
  return 0;
}

class GameDataReporter {
  private childId: string;
  private gameId: string;
  
  constructor(childId: string, gameId: string) {
    this.childId = childId;
    this.gameId = gameId;
  }
  
  // 游戏开始
  reportStart(config: any) {
    this.send('game_start', {
      childId: this.childId,
      gameId: this.gameId,
      config,
      timestamp: Date.now(),
    });
  }
  
  // 游戏结束
  reportEnd(result: any) {
    this.send('game_end', {
      childId: this.childId,
      gameId: this.gameId,
      result,
      timestamp: Date.now(),
    });
  }
  
  // 完整数据
  reportComplete(data: any) {
    this.send('game_complete', data);
  }
  
  // 发送数据
  private send(eventName: string, data: any) {
    // 实际实现中会调用API
    console.log(`[Report] ${eventName}:`, data);
  }
}
```
