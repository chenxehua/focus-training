### 1.3 儿童友好色彩原则

```
┌─────────────────────────────────────────────────────────────┐
│              儿童友好色彩设计原则                           │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  1. 高对比度                                                │
│     - 文字与背景对比度 ≥ 4.5:1                             │
│     - 重要信息对比度 ≥ 7:1                                │
│                                                             │
│  2. 色彩明快不刺眼                                          │
│     - 避免纯黑(#000000)和纯白(#FFFFFF)直接组合             │
│     - 使用柔和的背景色而非纯白                               │
│     - 高饱和色不超过画面30%                                 │
│                                                             │
│  3. 色彩语义一致性                                          │
│     - 绿色 = 正确/成功                                      │
│     - 红色 = 错误（柔和红#FF8A80，避免警示红）             │
│     - 黄色 = 目标/高亮                                     │
│     - 紫色 = 品牌/主操作                                    │
│                                                             │
│  4. 无障碍设计                                               │
│     - 不依赖颜色传达信息                                    │
│     - 配合图标/形状区分元素                                  │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

// 方格尺寸自适应计算
function calculateGridLayout(screenWidth: number, gridSize: number) {
  const pagePadding = 24;        // 页面边距
  const gridPadding = 0;         // 方格容器内边距
  const gridGap = getGridGap(gridSize);  // 方格间距
  
  // 可用宽度
  const availableWidth = screenWidth - (pagePadding * 2);
  
  // 计算方格尺寸（减去间距）
  const totalGapWidth = (gridSize - 1) * gridGap;
  const cellSize = Math.floor((availableWidth - totalGapWidth) / gridSize);
  
  // 最终方格容器宽度
  const gridWidth = cellSize * gridSize + totalGapWidth;
  
  return {
    cellSize,
    gridWidth,
    gridGap,
    fontSize: getFontSize(gridSize, cellSize),
  };
}

      6: 76,
      7: 66,
    },
  },

// 计算实际尺寸
function getResponsiveCellSize(gridSize: number, screenWidth: number): number {
  let breakpoints;
  
  if (screenWidth >= 800) {
    breakpoints = BREAKPOINTS.tabletLandscape.cellSizes;
  } else if (screenWidth >= 600) {
    breakpoints = BREAKPOINTS.tablet.cellSizes;
  } else if (screenWidth > 375) {
    breakpoints = BREAKPOINTS.largeMobile.cellSizes;
  } else {
    breakpoints = BREAKPOINTS.mobile.cellSizes;
  }
  
  return breakpoints[gridSize] || 60;
}
