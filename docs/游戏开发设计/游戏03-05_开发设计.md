### 4.3 移动算法

```typescript
// 目标移动轨迹
interface TargetMotion {
  startX: number;
  startY: number;
  path: { x: number; y: number; time: number }[];
  speed: number;
  shape: 'circle' | 'star' | 'heart';
  color: string;
}

// 生成随机路径
function generateRandomPath(
  startX: number,
  startY: number,
  duration: number,
  speed: number,
  bounds: { width: number; height: number }
): { x: number; y: number; time: number }[] {
  const path: { x: number; y: number; time: number }[] = [{ x: startX, y: startY, time: 0 }];
  let currentX = startX;
  let currentY = startY;
  const interval = 100; // 采样间隔
  
  for (let t = interval; t < duration; t += interval) {
    // 随机方向移动
    const angle = Math.random() * Math.PI * 2;
    const distance = speed * (interval / 1000);
    let newX = currentX + Math.cos(angle) * distance;
    let newY = currentY + Math.sin(angle) * distance;
    
    // 边界反弹
    if (newX < 0 || newX > bounds.width) newX = currentX;
    if (newY < 0 || newY > bounds.height) newY = currentY;
    
    currentX = newX;
    currentY = newY;
    path.push({ x: currentX, y: currentY, time: t });
  }
  
  return path;
}
```

```typescript
interface RhythmResult {
  perfect: number;    // 完美(±50ms)
  good: number;       // 良好(±100ms)
  ok: number;         // 一般(±tolerance)
  miss: number;       // 错过
}

function calculateRhythmScore(result: RhythmResult, difficulty: number): number {
  const total = result.perfect + result.good + result.ok + result.miss;
  const accuracy = (result.perfect * 100 + result.good * 70 + result.ok * 40) / (total * 100);
  const baseScore = accuracy * 100;
  const perfectBonus = result.perfect * 10;
  const difficultyFactor = difficulty * 0.2 + 0.5;
  
  return Math.round((baseScore + perfectBonus) * difficultyFactor);
}
```

```typescript
// 使用递归回溯算法生成迷宫
function generateMaze(size: number, density: number): MazeCell[][] {
  const maze: MazeCell[][] = [];
  
  // 初始化
  for (let y = 0; y < size; y++) {
    maze[y] = [];
    for (let x = 0; x < size; x++) {
      maze[y][x] = {
        type: Math.random() < density ? 'wall' : 'path',
        visited: false,
      };
    }
  }
  
  // 确保起点和终点是路径
  maze[0][0].type = 'path';
  maze[size-1][size-1].type = 'path';
  
  // 递归回溯创建路径
  carvePaths(maze, 0, 0, size);
  
  // 添加一些随机路径增加可玩性
  addRandomPaths(maze, density);
  
  return maze;
}

function carvePaths(maze: MazeCell[][], x: number, y: number, size: number) {
  const directions = shuffleArray([
    [0, -2], [0, 2], [-2, 0], [2, 0]
  ]);
  
  for (const [dx, dy] of directions) {
    const nx = x + dx;
    const ny = y + dy;
    
    if (nx > 0 && nx < size-1 && ny > 0 && ny < size-1 && !maze[ny][nx].visited) {
      maze[y + dy/2][x + dx/2].type = 'path';
      maze[ny][nx].type = 'path';
      maze[ny][nx].visited = true;
      carvePaths(maze, nx, ny, size);
    }
  }
}
```

### 9.3 星星移动算法

```typescript
// 生成星星移动路径
function generateStarPath(
  startX: number,
  startY: number,
  duration: number,
  smoothness: number
): StarPath[] {
  const path: StarPath[] = [];
  const points = Math.floor(duration / 16); // 60fps
  const controlPoints: { x: number; y: number }[] = [];
  
  // 生成随机控制点
  for (let i = 0; i <= 3; i++) {
    controlPoints.push({
      x: startX + (Math.random() - 0.5) * 200,
      y: startY + (Math.random() - 0.5) * 300,
    });
  }
  controlPoints[0] = { x: startX, y: startY };
  controlPoints[3] = {
    x: startX + (Math.random() - 0.5) * 300,
    y: startY + (Math.random() - 0.5) * 400,
  };
  
  // 使用贝塞尔曲线生成平滑路径
  for (let t = 0; t <= 1; t += 1/points) {
    const point = bezierCurve(controlPoints, t);
    path.push({
      x: point.x,
      y: point.y,
      time: t * duration,
    });
  }
  
  return path;
}

// 二次贝塞尔曲线
function bezierCurve(controlPoints: { x: number; y: number }[], t: number): { x: number; y: number } {
  const [p0, p1, p2, p3] = controlPoints;
  const mt = 1 - t;
  const mt2 = mt * mt;
  const t2 = t * t;
  
  return {
    x: mt2 * mt * p0.x + 3 * mt2 * t * p1.x + 3 * mt * t2 * p2.x + t2 * t * p3.x,
    y: mt2 * mt * p0.y + 3 * mt2 * t * p1.y + 3 * mt * t2 * p2.y + t2 * t * p3.y,
  };
}
```
